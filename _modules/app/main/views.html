[docs]def gagepage(gid=None, slug=None):
    """**/gage/<slug>/**

    Individual gage page
    """
    if slug is None:
        gage = Gage.query.get_or_404(gid)
    else:
        gage = Gage.query.filter_by(slug=slug).first_or_404()
    return render_template('gage.html', Gage=Gage, gage=gage)
