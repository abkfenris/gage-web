"""

Endpoints:
----------

- **/api/1.0/regions/<id>** - **GET** Detailed information about region *id*
- **/api/1.0/rivers/ - **GET** List all rivers

"""
from flask import jsonify, request, g, abort, url_for, current_app
from .. import db
from ..models import River
from . import api

@api.route('/rivers/')

[docs]def get_rivers():
	"""
	List all rivers
	
	Example response: ::
	
		{ "count": 7,
		  "next": "http://riverflo.ws/api/1.0/sensors/?page=2",
		  "prev": null,
		  "rivers": [
		    { "id": 1,
		      "name": "Kennebec",
		      "url": "http://riverflo.ws/api/1.0/rivers/1"
		    },
		    { "id": 2,
		      "name": "Androscoggin",
		      "url": "http://riverflo.ws/api/1.0/rivers/2"
		    }
		  ]
		}
	"""
	page = request.args.get('page', 1, type=int)
	pagination = River.query.paginate(page, per_page=current_app.config['API_GAGES_PER_PAGE'], error_out=False)
	rivers = pagination.items
	prev = None
	if pagination.has_prev:
		prev = url_for('.get_sensors', page=page-1, _external=True)
	next = None
	if pagination.has_next:
		next = url_for('.get_sensors', page=page+1, _external=True)
	return jsonify({
		'rivers': [river.to_json() for river in rivers],
		'prev': prev,
		'next': next,
		'count': pagination.total
	})
